
#include <iostream>
#include <cstdlib>
#include <iomanip>
using namespace std;


int main()
{
    
    int n;
    cin>>n;
    
    int **tab=new int*[n];
    for(int i=0; i<n; i++){
        tab[i]=new int [n];
    }
    
    int liczba=1;
    for(int i=0; i<n; i++){
        for (int j=0; j<n; j++){
            tab[i][j]=liczba++;
            
        }
    }
    
    for(int i=0; i<n; i++){
        for (int j=0; j<n; j++){
            cout<<setw(3)<<tab[j][i];
            
        }
        cout<<endl;
    }
    
    int suma=0;
    for(int i=0; i<n; i++){
        for (int j=0; j<n; j++){
            if(i==0||i==n-1||j==0||j==n-1){
                suma+=tab[i][j];
            }
        }
    }
    cout<<endl<<suma;
    
    for(int i=0; i<n; i++){
        delete[] tab[i];
    }
    delete[] tab;
    
    
    return suma;
}
