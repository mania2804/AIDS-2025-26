#include <iostream>

using namespace std;

int main()
{
    int a=6;
    
    for(int i=0; i<a/2; i++){
        for(int j=0; j<a; j++){
            if(i==j||j==a-i-1)
            cout<<"*";
            else 
            cout<<" ";
        }
        cout<<endl;
    }
    
    
    return 0;
}
